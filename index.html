// PARTE 4 — O CARTÃO MAGNÉTICO (efeito 3D)

# Quando o mouse passa por cima, o cartão se inclina # seguindo o cursor, como se fosse um objeto físico. mouse mexeu → calcula distância do centro → inclina o cartão nesse ângulo (máximo ajustável) → adiciona um brilho que segue o dedo/mouse # ↓ escolha o ângulo máximo e teste no cartãozinho ao lado:
passe o mouse ← o cartãozinho exagera o ângulo p/ ficar nítido
código real · JavaScript
cardWrap.addEventListener('mousemove', e => {
  const dx = (e.clientX - cx) / (w / 2);   // -1 ... +1
  const dy = (e.clientY - cy) / (h / 2);
  card.style.transform =
    `rotateX(${-dy * tiltMax}deg) rotateY(${dx * tiltMax}deg)`;
});
